# 搭建博客

# VuePress 搭建博客

# 安装步骤如下

  1. 全局安装
 npm install -g vuepress
  1. 创建新项目目录
  mkdir docs
  1. 初始化项目
 npm init -y
  1. 配置package.json 添加下面这两行
{
 "scripts": {
   "docs:dev": "vuepress dev docs",
   "docs:build": "vuepress build docs"
 }
}
  1. 配置config.js
module.exports = {
 title: 'WEB',
 description: 'Just playing around',

 themeConfig: {
   nav: [
     { text: '首页', link: '/' },
{
text:'笔记',link:'/js/',
},
     {
       text: 'js',
       items: [
         //下拉导航
         { text: 'JS基础', link: '/js/' },
         { text: 'js高级', link: '/jsg/' }
       ]
     }
   ],
# 注意事项：
  • 和图标/图片等静态资源相关的 其 '/' 默认指向的是 docs/.vuepress/public/
  • 和侧边栏/导航栏相关的地址配置 其 '/' 默认指向的是 docs/
  • 侧边栏/导航栏指向.md文件的需要先建立相关.md文件，不然会报404或者页面空白
  1. 配置导航 .vuepress/config
 themeConfig: {
 nav: [
   { text: '首页', link: '/' },
   { text: 'web', link: '/web/' }, //导航
   {
     //可以下拉导航
     text: 'js',
     items: [
       { text: 'JS基础', link: '/js/' },
       { text: 'js高级', link: '/jsg/' }
     ]
   }
 ]
}

# 7.发布GitHub

在文件根目录下建立 deploy.sh 文件 放以下代码

#!/usr/bin/env sh

# 确保脚本抛出遇到的错误
set -e

# 生成静态文件
npm run build

# 进入生成的文件夹
cd docs/.vuepress/dist

# 如果是发布到自定义域名
# echo 'www.example.com' > CNAME

git init
git add -A
git commit -m 'deploy'

# 如果发布到 https://<USERNAME>.github.io
git push -f git@github.com:lq110/lq110.github.io.git master

# 如果发布到 https://<USERNAME>.github.io/<REPO>
# git push -f git@github.com:<USERNAME>/<REPO>.git master:gh-pages

cd -

  1. git 仓库起名必须用 github 名开头

  2. 发布时候的命令 在 git 中运行

  npm run deploy
